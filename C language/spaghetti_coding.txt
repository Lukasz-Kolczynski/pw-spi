int getMaxInt(int *tab, unsigned int size){
    int i;
    int max = tab[0];
    
    for(i=0; i<size; i++){
        if (tab[i] > max){
            max = tab[i];
            
        }
        
    }
    return max;

}

double getMaxDouble(double *tab, unsigned int size){
    int i;
    double max = tab[0];
    
    for(i=0; i<size; i++){
        if (tab[i] > max){
            max = tab[i];
            
        }
        
    }
    return max;

}
